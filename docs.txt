prefix = /file_storage/api/v1.0/
POST prefix+'users' {'username':username, 'password':password} -- add user
DELETE prefix+'user/'+username -- delete user
PUT prefix+'user/'+username {'password':new_password} -- update user
GET prefix+'user/' -- get all usernames

GET prefix+'root' -- get all directories
POST prefix+'root' {'directory_name':dir_name} -- add directory
PUT prefix+'root/' + dir_name {'directory_name':new_dir_name} -- update dir_name
DELETE prefix+'root/' + dir_name -- delete directory

GET prefix+'root/' + dir_name -- get all files in this directory
POST prefix+'root/' + dir_name {'filename':file_name,'filecontent':filecontent} -- add file
PUT prefix+'root/' + dir_name + '/' + file_name {'filecontent':filecontent} -- update file content
DELETE prefix+'root/' + dir_name + '/' + file_name -- delete file

GET prefix+'root/' + dir_name + '/' + file_name -- get file content
